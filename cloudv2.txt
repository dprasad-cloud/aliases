echo -e "Shortcut for checking resource usage on nodes (ex: topnode ip-10-201-50-219)"
topnode(){
  node="$1"
  echo -e "POD_NAME\tCPU\tMEM"

  tmp_pods="$(mktemp)" || return 1
  tmp_top="$(mktemp)" || { rm -f "$tmp_pods"; return 1; }

  kubectl get pods -A -o wide | grep "$node" | awk '{print $2}' | sort > "$tmp_pods"
  kubectl top pods -A --no-headers | awk '{print $2, $3, $4}' | sort > "$tmp_top"

  join -1 1 -2 1 "$tmp_pods" "$tmp_top" | column -t | sort -rnk2

  rm -f "$tmp_pods" "$tmp_top"
}
# Prefer functions over aliases so these work in non-interactive shells too.
topnodes_90(){
  kubectl top nodes | awk '$0=="" || /'"$(date +%Y)"'/ || $3+0 >= 90 || $5+0 > 90 || $2 == "<unknown>"'
}

topnodes_95(){
  kubectl top nodes | awk '$0=="" || /'"$(date +%Y)"'/ || $3+0 >= 95 || $5+0 > 95 || $2 == "<unknown>"'
}

topnodes_90status(){
  tmp="$(mktemp)" || return 1
  topnodes90 | awk '{print $1}' > "$tmp"
  kubectl get nodes | grep -Ff "$tmp"
  rm -f "$tmp"
}

topnodes_95status(){
  tmp="$(mktemp)" || return 1
  topnodes95 | awk '{print $1}' > "$tmp"
  kubectl get nodes | grep -Ff "$tmp"
  rm -f "$tmp"
}

# Commonly used fpod alias for finding pods in error states or with high restart counts
export errstates='term|err|crash|pend|evict|pull|creating|init:|oom|(0/|1/2|2/3).*Running|\s([1-9][0-9]|[0-9]{3,})\s'
export commonkafka="unset JMX_PORT; bin/kafka-cluster.sh cluster-id --bootstrap-server common-kafka:9092"
export defaultkafka="unset JMX_PORT; bin/kafka-cluster.sh cluster-id --bootstrap-server default-kafka:9092"
export nvokafka="unset JMX_PORT; bin/kafka-cluster.sh cluster-id --bootstrap-server nvo-kafka:9092"

#Usage (ferr or ferr "search_key")
unalias ferr 2>/dev/null || true
ferr() {
    if [ -z "$1" ]; then
        fpod $errstates
    else
        fpod $errstates | grep -iE "$1"
    fi
}

loop() { while true; do date; eval "$@"; echo -e "\n"; sleep 5; done; }
loop15() { while true; do date; eval "$@"; echo -e "\n"; sleep 15; done; }

ups () {
    # Define the target file name inside the extracted directory
    file="cloudv2.txt"

    # 1. Download the repository archive and save it to /root/main.zip
    wget -O /root/main.zip https://github.com/dprasad-cloud/aliases/archive/refs/heads/main.zip

    # 2. Unzip the archive to /root, overwriting existing files (-o)
    unzip -o /root/main.zip -d /root

    # 3. Source the specific file (cloudv2.txt) from the extracted folder (aliases-main)
    # This executes the commands/aliases defined in cloudv2.txt in the current shell session.
    source /root/aliases-main/$file
}
