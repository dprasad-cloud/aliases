hmcat_all() {
  local pod_pattern="$1"
  local search_term="$2"

  if [ -z "$pod_pattern" ] || [ -z "$search_term" ]; then
    echo "Usage: search_pod_logs <pod_pattern> <search_term>"
    return 1
  fi

  for pod in $(fpod "$pod_pattern" | awk '{print $2}'); do
    echo "-----------Processing pod: $pod"
    # Assuming hmcat handles the pod argument and outputs to stdout
    hmcat "$pod" | grep -iE "$search_term"
  done
}

echo -e "Shortcut for checking resource usage on nodes (ex: topnode ip-10-201-50-219)"
topnode(){
  node="$1"
  echo -e "POD_NAME\tCPU\tMEM"

  tmp_pods="$(mktemp)" || return 1
  tmp_top="$(mktemp)" || { rm -f "$tmp_pods"; return 1; }

  kubectl get pods -A -o wide | grep "$node" | awk '{print $2}' | sort > "$tmp_pods"
  kubectl top pods -A --no-headers | awk '{print $2, $3, $4}' | sort > "$tmp_top"

  join -1 1 -2 1 "$tmp_pods" "$tmp_top" | column -t | sort -rnk2

  rm -f "$tmp_pods" "$tmp_top"
}
# Prefer functions over aliases so these work in non-interactive shells too.
#
# Usage:
#   topnodes 90
#   topnodes 95
#   topnodes_status 90
#   topnodes_status 95
topnodes() {
  local input="$1"
  if [ -z "$input" ]; then
    echo "Usage: topnodes [+|-]threshold" >&2
    return 2
  fi

  local op=">="
  local thr="$input"

  if [[ "$input" == -* ]]; then
    op="<"
    thr="${input#-}"
  elif [[ "$input" == +* ]]; then
    op=">="
    thr="${input#+}"
  fi

  kubectl top nodes | awk -v op="$op" -v thr="$thr" '
    $1 == "NAME" { next }  # Skip the header row
    $0=="" || /'$(date +%Y)'/ || $2 == "<unknown>" || \
    (op == ">=" && ($3+0 >= thr || $5+0 >= thr)) || \
    (op == "<"  && ($3+0 <  thr || $5+0 <  thr))'
}

function topnodes_status(){
  threshold="$1"
  if [ -z "$threshold" ]; then
    echo "Usage: topnodes_status <cpu_threshold_percent>" >&2
    return 2
  fi

  tmp="$(mktemp)" || return 1
  topnodes "$threshold" | awk '{print $1}' > "$tmp"
  kubectl get nodes | grep -Ff "$tmp"
  rm -f "$tmp"
}

topnodes_pods() {
    local threshold=$1
    local limit=${2:-100}
    local adjusted_limit=$((limit + 1))

    for node in $(topnodes "$threshold" | awk '{print $1}'); do
        echo "--- Node: $node ---"
        # Pipe to 'column -t' to normalize spacing
        topnode "$node" | head -n "$adjusted_limit" | column -t
        echo ""
    done
}

topcon() {
    local search_term=""
    local sort_col=""

    # Parse arguments
    while [ "$#" -gt 0 ]; do
        case "$1" in
            -r)
                sort_col="cpu"
                shift
                ;;
            -m)
                sort_col="mem"
                shift
                ;;
            *)
                search_term="$1"
                shift
                ;;
        esac
    done

    # Default to CPU sort if no flag is provided
    if [ -z "$sort_col" ]; then
        sort_col="cpu"
    fi

    # 1. Get metrics and strip header
    # 2. Filter using your grepie
    # 3. Use awk to append the target value to the front of the line,
    #    sort it numerically, and then strip that temporary value away.
    kubectl top pod -A --containers | tail -n +2 | grepie "$search_term" | awk -v mode="$sort_col" '
        {
            # NF is the last column (Memory), NF-1 is second-to-last (CPU)
            val = (mode == "cpu") ? $(NF-1) : $NF;

            # Strip units (m, Mi, Gi) so sort -rn works flawlessly
            gsub(/[a-zA-Z]/, "", val);

            print val "\t" $0;
        }
    ' | sort -rn | cut -f2-
}
